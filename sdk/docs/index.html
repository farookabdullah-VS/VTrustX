ğŸ  Overview

The VTrustX SDK provides native client libraries for three platforms. All platforms share the same REST API and survey data model, giving you a consistent developer experience everywhere.

class MainActivity : ComponentActivity() {

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        client = VTrustXClient(context = this)

        lifecycleScope.launch {
            // Login

                            }

                    }
                }
            }
        }
    }

    private fun submitAnswers(formId: String, answers: Map<String, Any>) {
        lifecycleScope.launch {
            client.createSubmission(formId, answers).onSuccess {
                Log.d("SDK", "Submitted successfully!")
            }
        }
    }
}
import SwiftUI
import VTrustX

struct ContentView: View {
    @State private var survey: NativeSurveyDefinition?
    @State private var formId: String = ""

    var body: some View {
        Group {
            if let definition = survey {
                SurveyView(

                    onComplete: { answers in

                    }
                )
            } else {
                ProgressView("Loading surveyâ€¦")
                    .task { await loadSurvey() }
            }
        }
    }

    func loadSurvey() async {
        do {
            // Login first
            try await client.login(username: "admin", password: "password")

            // Load survey by slug
            let form = try await client.getFormBySlug(slug: "my-nps-survey")
            formId = form.id
            survey = form.definition
        } catch {
            print("Error:", error)
        }
    }

    func submit(formId: String, answers: [String: Any]) async {
        do {
            try await client.createSubmission(formId: formId, data: answers)
            print("Submitted!")
        } catch {
            print("Error:", error)
        }
    }
}

ğŸ” Authentication

The SDK supports two authentication methods. The server returns a Bearer token when authenticating from a mobile or API-key context. Cookie-based auth is used in browser environments automatically.

Token storage: On Android, tokens are stored in EncryptedSharedPreferences (AES-256-GCM). On iOS, tokens are stored in the Keychain. In the JS SDK, token state is held in memory by the client instance.

Every survey is described by a NativeSurveyDefinition â€” a JSON document that all three SDK platforms can parse and render natively.

NativeSurveyDefinition

interface NativeSurveyDefinition {
    id: string;
    title: string;
    description?: string;
    theme: SurveyTheme;
    screens: SurveyScreen[];
    logic?: LogicRule[];
}

interface SurveyTheme {
    primaryColor: string;       // hex e.g. "#2563eb"
    backgroundColor: string;
    typography?: 'system' | 'serif' | 'sans-serif' | 'monospace';
    borderRadius?: number;      // default: 8
    darkMode?: boolean;
}

interface SurveyScreen {
    id: string;
    title?: string;
    components: SurveyComponent[];
}

interface SurveyComponent {
    id: string;
    type: 'text' | 'multiline_text' | 'number' | 'rating' | 'nps'
        | 'single_choice' | 'multiple_choice' | 'date' | 'slider' | 'file_upload';
    label: string;
    required?: boolean;
    options?: Array<{ value: string; label: string }>;
    min?: number;
    max?: number;
    step?: number;
    placeholder?: string;
}
data class NativeSurveyDefinition(
    val id: String,
    val title: String,
    val description: String?,
    val theme: SurveyTheme,
    val screens: List<SurveyScreen>,
    val logic: List<LogicRule>?
)

data class SurveyTheme(
    val primaryColor: String,       // hex e.g. "#2563eb"

    val typography: String?,
    val borderRadius: Double?,
    val darkMode: Boolean?
)

data class SurveyScreen(
    val id: String,
    val title: String?,
    val components: List<SurveyComponent>
)

data class SurveyComponent(
    val id: String,

ğŸ§  Logic & Validation

The survey engine handles skip logic, conditional branching, and field validation on the client side â€” no network round-trips required.

import { SurveyEngine } from '@vtrustx/sdk';

const engine = new SurveyEngine(definition);

// Validate a single component answer

// Navigate with skip logic

if (nextId === null) {
    // Survey finished
} else {
    goToScreen(nextId);
}

// â”€â”€â”€ Logic Rule example (in survey definition) â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
// "If NPS score < 7  â†’ skip to detractor follow-up screen"
const rule: LogicRule = {
    trigger: {
        questionId: 'nps_score',
        operator: 'less_than',
        value: 7
    },
    action: 'skip_to',
    target: 'screen-detractor'
};
import com.vtrustx.sdk.engine.SurveyEngine

val engine = SurveyEngine(definition)

// Validate current screen before advancing
val errors = engine.validateScreen(currentScreen, answers)

    errors.forEach { (id, msg) -> showError(id, msg) }
    return
}

// Determine next screen (handles skip logic automatically)
val nextId = engine.getNextScreen(
    currentScreenId = currentScreen.id,
    answers = answers
)

if (nextId == null) {
    onComplete(answers)          // Survey finished
} else {
    navigateToScreen(nextId)
}

@Entity(tableName = "pending_submissions")
data class PendingSubmission(
    @PrimaryKey(autoGenerate = true) val id: Int = 0,
    val formId: String,
    val data: String,       // JSON-encoded Map<String, Any>
    val timestamp: Long = System.currentTimeMillis(),
    val synced: Boolean = false
)

// Save locally when offline
suspend fun saveOffline(formId: String, answers: Map<String, Any>) {
    val row = PendingSubmission(
        formId = formId,
        data   = Gson().toJson(answers)
    )
    db.pendingDao().insert(row)
}

// Sync when back online
suspend fun syncPending(client: VTrustXClient) {
    db.pendingDao().getUnsynced().forEach { row ->
        val answers = Gson().fromJson(row.data, Map::class.java)
                          as Map<String, Any>
        client.createSubmission(row.formId, answers)
            .onSuccess { db.pendingDao().markSynced(row.id) }
    }
}
struct PendingSubmission: Codable {
    let formId: String
    let data: [String: AnyCodable]
    let timestamp: Date
    var synced: Bool = false
}

// Save locally when offline
func saveOffline(formId: String, answers: [String: Any]) {
    var queue = loadQueue()
    queue.append(PendingSubmission(
        formId: formId,
        data: answers.mapValues { AnyCodable($0) },
        timestamp: Date()
    ))
    saveQueue(queue)
}

// Sync when back online
func syncPending(client: VTrustXClient) async {
    var queue = loadQueue()
    for (i, item) in queue.enumerated() where !item.synced {
        do {
            try await client.createSubmission(
                formId: item.formId,
                data: item.data.mapValues { $0.value }
            )
            queue[i].synced = true
        } catch { continue }
    }
    saveQueue(queue)
}

func loadQueue() -> [PendingSubmission] {
    guard let data = UserDefaults.standard.data(forKey: "pendingQueue"),
          let queue = try? JSONDecoder().decode([PendingSubmission].self, from: data)
    else { return [] }
    return queue
}

func saveQueue(_ queue: [PendingSubmission]) {
    if let data = try? JSONEncoder().encode(queue) {
        UserDefaults.standard.set(data, forKey: "pendingQueue")
    }
}
